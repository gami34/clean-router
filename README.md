## Install

```bash
npm install --save clean-router
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'clean-router'
import 'clean-router/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
